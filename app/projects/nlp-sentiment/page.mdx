## Technical Architecture

### Model Selection and Fine-tuning

I experimented with several pre-trained models and settled on a fine-tuned BERT-base model with custom classification heads:

<CodeBlock language="python" filename="model_architecture.py">
{`import torch
import torch.nn as nn
from transformers import BertModel, BertTokenizer

class SentimentBERT(nn.Module):
    def __init__(self, bert_model_name='bert-base-uncased', num_classes=3, dropout_rate=0.3):
        super(SentimentBERT, self).__init__()
        
        # Load pre-trained BERT
        self.bert = BertModel.from_pretrained(bert_model_name)
        self.tokenizer = BertTokenizer.from_pretrained(bert_model_name)
        
        # Classification head
        self.dropout = nn.Dropout(dropout_rate)
        self.classifier = nn.Sequential(
            nn.Linear(self.bert.config.hidden_size, 512),
            nn.ReLU(),
            nn.Dropout(dropout_rate),
            nn.Linear(512, 256),
            nn.ReLU(),
            nn.Dropout(dropout_rate),
            nn.Linear(256, num_classes)
        )
        
        # Initialize weights
        self._init_weights()
    
    def _init_weights(self):
        for module in self.classifier:
            if isinstance(module, nn.Linear):
                nn.init.xavier_uniform_(module.weight)
                nn.init.zeros_(module.bias)
    
    def forward(self, input_ids, attention_mask, token_type_ids=None):
        # BERT forward pass
        outputs = self.bert(
            input_ids=input_ids,
            attention_mask=attention_mask,
            token_type_ids=token_type_ids
        )
        
        # Use [CLS] token representation
        pooled_output = outputs.pooler_output
        
        # Apply dropout and classification
        pooled_output = self.dropout(pooled_output)
        logits = self.classifier(pooled_output)
        
        return logits`}
</CodeBlock>

### Data Preprocessing Pipeline

The preprocessing pipeline handles various text formats and implements advanced tokenization strategies:

<CodeBlock language="python" filename="preprocessing.py">
{`import re
import pandas as pd
from transformers import BertTokenizer
from torch.utils.data import Dataset, DataLoader

class TextPreprocessor:
    def __init__(self, tokenizer_name='bert-base-uncased'):
        self.tokenizer = BertTokenizer.from_pretrained(tokenizer_name)
        
    def clean_text(self, text):
        """Advanced text cleaning for sentiment analysis"""
        # Remove URLs
        text = re.sub(r'http\S+|www.\S+', '[URL]', text)
        
        # Handle mentions and hashtags
        text = re.sub(r'@\w+', '[USER]', text)
        text = re.sub(r'#(\w+)', r'\1', text)
        
        # Normalize whitespace
        text = re.sub(r'\s+', ' ', text.strip())
        
        # Handle repeated characters (e.g., "sooooo" -> "soo")
        text = re.sub(r'(.)\1{2,}', r'\1\1', text)
        
        return text
    
    def tokenize_batch(self, texts, max_length=512):
        """Tokenize a batch of texts"""
        cleaned_texts = [self.clean_text(text) for text in texts]
        
        encoded = self.tokenizer(
            cleaned_texts,
            truncation=True,
            padding=True,
            max_length=max_length,
            return_tensors='pt',
            return_attention_mask=True
        )
        
        return encoded

class SentimentDataset(Dataset):
    def __init__(self, texts, labels, tokenizer, max_length=512):
        self.texts = texts
        self.labels = labels
        self.tokenizer = tokenizer
        self.max_length = max_length
        self.preprocessor = TextPreprocessor()
    
    def __len__(self):
        return len(self.texts)
    
    def __getitem__(self, idx):
        text = str(self.texts[idx])
        label = self.labels[idx]
        
        # Preprocess text
        text = self.preprocessor.clean_text(text)
        
        # Tokenize
        encoding = self.tokenizer(
            text,
            truncation=True,
            padding='max_length',
            max_length=self.max_length,
            return_tensors='pt'
        )
        
        return {
            'text': text,
            'input_ids': encoding['input_ids'].flatten(),
            'attention_mask': encoding['attention_mask'].flatten(),
            'label': torch.tensor(label, dtype=torch.long)
        }`}
</CodeBlock>

## Training Strategy

### Multi-stage Training Process

I implemented a sophisticated training strategy with multiple stages:

1. **Warmup Phase**: Gradual learning rate increase
2. **Fine-tuning**: BERT layers with differential learning rates
3. **Regularization**: Advanced dropout and weight decay
4. **Validation**: K-fold cross-validation for robust evaluation

<CodeBlock language="python" filename="training.py">
{`import torch
import torch.nn as nn
from torch.optim import AdamW
from transformers import get_linear_schedule_with_warmup
from sklearn.metrics import accuracy_score, classification_report
import wandb

class SentimentTrainer:
    def __init__(self, model, train_loader, val_loader, device='cuda'):
        self.model = model.to(device)
        self.train_loader = train_loader
        self.val_loader = val_loader
        self.device = device
        
        # Initialize optimizer with differential learning rates
        self.optimizer = self._setup_optimizer()
        
        # Setup learning rate scheduler
        total_steps = len(train_loader) * 5  # 5 epochs
        self.scheduler = get_linear_schedule_with_warmup(
            self.optimizer,
            num_warmup_steps=total_steps * 0.1,
            num_training_steps=total_steps
        )
        
        # Loss function with class weights
        self.criterion = nn.CrossEntropyLoss()
        
        # Initialize Weights & Biases for tracking
        wandb.init(project="sentiment-analysis")
    
    def _setup_optimizer(self):
        """Setup optimizer with differential learning rates"""
        # Different learning rates for different parts of the model
        bert_params = []
        classifier_params = []
        
        for name, param in self.model.named_parameters():
            if 'bert' in name:
                bert_params.append(param)
            else:
                classifier_params.append(param)
        
        optimizer = AdamW([
            {'params': bert_params, 'lr': 2e-5},
            {'params': classifier_params, 'lr': 1e-4}
        ], weight_decay=0.01)
        
        return optimizer
    
    def train_epoch(self):
        self.model.train()
        total_loss = 0
        predictions = []
        true_labels = []
        
        for batch in self.train_loader:
            # Move batch to device
            input_ids = batch['input_ids'].to(self.device)
            attention_mask = batch['attention_mask'].to(self.device)
            labels = batch['label'].to(self.device)
            
            # Forward pass
            self.optimizer.zero_grad()
            logits = self.model(input_ids, attention_mask)
            loss = self.criterion(logits, labels)
            
            # Backward pass
            loss.backward()
            torch.nn.utils.clip_grad_norm_(self.model.parameters(), 1.0)
            self.optimizer.step()
            self.scheduler.step()
            
            total_loss += loss.item()
            
            # Collect predictions for metrics
            preds = torch.argmax(logits, dim=1)
            predictions.extend(preds.cpu().numpy())
            true_labels.extend(labels.cpu().numpy())
        
        # Calculate metrics
        accuracy = accuracy_score(true_labels, predictions)
        avg_loss = total_loss / len(self.train_loader)
        
        return avg_loss, accuracy
    
    def validate(self):
        self.model.eval()
        total_loss = 0
        predictions = []
        true_labels = []
        
        with torch.no_grad():
            for batch in self.val_loader:
                input_ids = batch['input_ids'].to(self.device)
                attention_mask = batch['attention_mask'].to(self.device)
                labels = batch['label'].to(self.device)
                
                logits = self.model(input_ids, attention_mask)
                loss = self.criterion(logits, labels)
                
                total_loss += loss.item()
                
                preds = torch.argmax(logits, dim=1)
                predictions.extend(preds.cpu().numpy())
                true_labels.extend(labels.cpu().numpy())
        
        accuracy = accuracy_score(true_labels, predictions)
        avg_loss = total_loss / len(self.val_loader)
        
        return avg_loss, accuracy, predictions, true_labels`}
</CodeBlock>

{`import matplotlib.pyplot as plt
import seaborn as sns
from lime.lime_text import LimeTextExplainer
import numpy as np

class ModelInterpreter:
    def __init__(self, model, tokenizer):
        self.model = model
        self.tokenizer = tokenizer
        self.lime_explainer = LimeTextExplainer(class_names=['Negative', 'Neutral', 'Positive'])
    
    def visualize_attention(self, text, max_length=128):
        """Visualize attention weights for input text"""
        # Tokenize input
        inputs = self.tokenizer(
            text,
            truncation=True,
            padding='max_length',
            max_length=max_length,
            return_tensors='pt',
            return_attention_mask=True
        )
        
        # Get attention weights
        with torch.no_grad():
            outputs = self.model.bert(
                input_ids=inputs['input_ids'],
                attention_mask=inputs['attention_mask'],
                output_attentions=True
            )
        
        # Extract attention from last layer, first head
        attention = outputs.attentions[-1][0, 0].cpu().numpy()
        
        # Get tokens
        tokens = self.tokenizer.convert_ids_to_tokens(inputs['input_ids'][0])
        
        # Create visualization
        fig, ax = plt.subplots(figsize=(12, 8))
        sns.heatmap(
            attention[:len(tokens), :len(tokens)],
            xticklabels=tokens,
            yticklabels=tokens,
            cmap='Blues',
            ax=ax
        )
        ax.set_title('Attention Weights Visualization')
        plt.xticks(rotation=45)
        plt.yticks(rotation=0)
        plt.tight_layout()
        
        return fig
    
    def explain_prediction(self, text, num_features=10):
        """Generate LIME explanation for prediction"""
        def predict_fn(texts):
            predictions = []
            for text in texts:
                inputs = self.tokenizer(
                    text,
                    truncation=True,
                    padding='max_length',
                    max_length=512,
                    return_tensors='pt'
                )
                
                with torch.no_grad():
                    logits = self.model(
                        inputs['input_ids'],
                        inputs['attention_mask']
                    )
                    probs = torch.softmax(logits, dim=1).cpu().numpy()[0]
                
                predictions.append(probs)
            
            return np.array(predictions)
        
        # Generate explanation
        explanation = self.lime_explainer.explain_instance(
            text,
            predict_fn,
            num_features=num_features,
            num_samples=500
        )
        
        return explanation`}

### API Development

<CodeBlock language="python" filename="api.py">
{`from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
import torch
from transformers import BertTokenizer
import uvicorn
import time

app = FastAPI(title="Sentiment Analysis API", version="1.0.0")

# Load model and tokenizer
model = torch.load('sentiment_model.pth', map_location='cpu')
tokenizer = BertTokenizer.from_pretrained('bert-base-uncased')
model.eval()

class TextInput(BaseModel):
    text: str
    confidence_threshold: float = 0.8

class PredictionOutput(BaseModel):
    text: str
    sentiment: str
    confidence: float
    processing_time: float

@app.post("/predict", response_model=PredictionOutput)
async def predict_sentiment(input_data: TextInput):
    start_time = time.time()
    
    try:
        # Preprocess and tokenize
        inputs = tokenizer(
            input_data.text,
            truncation=True,
            padding='max_length',
            max_length=512,
            return_tensors='pt'
        )
        
        # Make prediction
        with torch.no_grad():
            logits = model(inputs['input_ids'], inputs['attention_mask'])
            probabilities = torch.softmax(logits, dim=1)
            confidence, predicted = torch.max(probabilities, 1)
        
        # Map prediction to sentiment
        sentiment_map = {0: 'Negative', 1: 'Neutral', 2: 'Positive'}
        sentiment = sentiment_map[predicted.item()]
        confidence_score = confidence.item()
        
        processing_time = time.time() - start_time
        
        return PredictionOutput(
            text=input_data.text,
            sentiment=sentiment,
            confidence=confidence_score,
            processing_time=processing_time
        )
    
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.get("/health")
async def health_check():
    return {"status": "healthy", "model": "loaded"}

if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=8000)`}
