## Dataset Information

### Data Characteristics

The dataset consists of high-quality photographs of plastic items with the following specifications:

**Image Parameters:**
- **Size**: 1920x1277 pixels
- **Resolution**: 300 DPI
- **Color Palette**: RGB 24 bits
- **File Format**: JPG

### Filename Convention

The dataset uses a systematic naming convention where each filename contains detailed information about the object:

**Format**: `[order_number] [a][b][c][d][e][f][g][h].jpg`

**Parameter Breakdown:**
- **First four digits**: Order number of the object
- **[a]**: Plastic type number  
- **[b]**: Color classification
- **[c]**: Type of lighting used
- **[d]**: Deformation level
- **[e]**: Dirt level
- **[f]**: Presence of screw cap or lid
- **[g]**: Presence of ring (characteristic for bottles with screw caps)
- **[h]**: Random position order number

### Plastic Type Classifications [a]

The dataset includes the following plastic type categories:

- **00**: Missing or unreadable identifier
- **01**: **PET** - Polyethylene Terephthalate
- **02**: **PE-HD** - High-Density Polyethylene  
- **03**: **PVC** - Polyvinyl Chloride
- **04**: **PE-LD** - Low-Density Polyethylene
- **05**: **PP** - Polypropylene
- **06**: **PS** - Polystyrene
- **07**: **Other** - Mixed or other plastic types

### Example Filename Analysis

**Example**: `0004 a01b05c2d0e1f0g1h1.jpg`

**Breakdown:**
- **0004**: Object number 4
- **a01**: PET plastic type
- **b05**: Color classification 5
- **c2**: Lighting type 2
- **d0**: No deformation
- **e1**: Dirt level 1
- **f0**: No screw cap or lid
- **g1**: Ring present
- **h1**: Position 1

**Note**: This particular example shows a PET plastic item that is missing or has an unreadable identifier despite being classified as type 01.

import torch.nn as nn
import torch.nn.functional as F
from torchvision import models
import cv2
import numpy as np

class PlasticClassificationHead(nn.Module):
    def __init__(self, input_features, num_classes=7):
        super().__init__()
        self.classifier = nn.Sequential(
            nn.Dropout(0.5),
            nn.Linear(input_features, 512),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(512, 256),
            nn.ReLU(),
            nn.Linear(256, num_classes)
        )
    
    def forward(self, x):
        return self.classifier(x)

class PlasticDetector(nn.Module):
    def __init__(self, num_classes=7, pretrained=True):
        super().__init__()
        
        # Backbone: ResNet50 for feature extraction
        self.backbone = models.resnet50(pretrained=pretrained)
        self.backbone_features = self.backbone.fc.in_features
        
        # Remove final classification layer
        self.backbone = nn.Sequential(*list(self.backbone.children())[:-1])
        
        # Classification head
        self.classifier = PlasticClassificationHead(self.backbone_features, num_classes)
        
        # Object detection head (simplified YOLO-style)
        self.detection_head = nn.Sequential(
            nn.Conv2d(2048, 1024, 3, padding=1),
            nn.ReLU(),
            nn.Conv2d(1024, 512, 3, padding=1),
            nn.ReLU(),
            nn.Conv2d(512, 5 + num_classes, 1)  # 5 = (x, y, w, h, confidence)
        )
        
        # Feature map dimensions (for 224x224 input)
        self.grid_size = 7
        self.num_classes = num_classes
        
    def forward(self, x, mode='detection'):
        # Extract features
        features = self.backbone(x)  # Shape: (batch, 2048, 7, 7)
        
        if mode == 'classification':
            # Global average pooling for classification
            pooled_features = F.adaptive_avg_pool2d(features, (1, 1))
            pooled_features = pooled_features.view(pooled_features.size(0), -1)
            classification_output = self.classifier(pooled_features)
            return classification_output
        
        elif mode == 'detection':
            # Object detection output
            detection_output = self.detection_head(features)
            
            batch_size = x.size(0)
            detection_output = detection_output.view(
                batch_size, 
                self.grid_size, 
                self.grid_size, 
                5 + self.num_classes
            )
            
            return detection_output
        
        elif mode == 'both':
            # Both classification and detection
            pooled_features = F.adaptive_avg_pool2d(features, (1, 1))
            pooled_features = pooled_features.view(pooled_features.size(0), -1)
            classification_output = self.classifier(pooled_features)
            
            detection_output = self.detection_head(features)
            detection_output = detection_output.view(
                batch_size, 
                self.grid_size, 
                self.grid_size, 
                5 + self.num_classes
            )
            
            return classification_output, detection_output

class PlasticLoss(nn.Module):
    def __init__(self, lambda_coord=5.0, lambda_noobj=0.5):
        super().__init__()
        self.lambda_coord = lambda_coord
        self.lambda_noobj = lambda_noobj
        self.mse_loss = nn.MSELoss(reduction='sum')
        self.ce_loss = nn.CrossEntropyLoss()
    
    def forward(self, predictions, targets, mode='detection'):
        if mode == 'classification':
            return self.ce_loss(predictions, targets)
        
        elif mode == 'detection':
            batch_size = predictions.size(0)
            grid_size = predictions.size(1)
            
            # Parse predictions
            pred_boxes = predictions[..., :4]      # (x, y, w, h)
            pred_conf = predictions[..., 4]        # confidence
            pred_class = predictions[..., 5:]      # class probabilities
            
            # Parse targets
            target_boxes = targets[..., :4]
            target_conf = targets[..., 4]
            target_class = targets[..., 5:].argmax(dim=-1)
            
            # Object mask (where objects exist)
            obj_mask = target_conf > 0
            noobj_mask = target_conf == 0
            
            # Coordinate loss (only for cells with objects)
            coord_loss = self.lambda_coord * self.mse_loss(
                pred_boxes[obj_mask], 
                target_boxes[obj_mask]
            )
            
            # Confidence loss
            conf_loss_obj = self.mse_loss(
                pred_conf[obj_mask], 
                target_conf[obj_mask]
            )
            
            conf_loss_noobj = self.lambda_noobj * self.mse_loss(
                pred_conf[noobj_mask], 
                target_conf[noobj_mask]
            )
            
            # Classification loss (only for cells with objects)
            if obj_mask.sum() > 0:
                class_loss = self.ce_loss(
                    pred_class[obj_mask], 
                    target_class[obj_mask]
                )
            else:
                class_loss = torch.tensor(0.0, device=predictions.device)
            
            total_loss = coord_loss + conf_loss_obj + conf_loss_noobj + class_loss
            
            return total_loss, {
                'coord_loss': coord_loss.item(),
                'conf_loss_obj': conf_loss_obj.item(),
                'conf_loss_noobj': conf_loss_noobj.item(),
                'class_loss': class_loss.item()
            }

def non_max_suppression(predictions, conf_threshold=0.5, iou_threshold=0.4):
    """Apply Non-Maximum Suppression to predictions"""
    
    # Convert predictions to bounding boxes
    boxes = []
    confidences = []
    class_ids = []
    
    grid_size = predictions.size(1)
    
    for i in range(grid_size):
        for j in range(grid_size):
            pred = predictions[0, i, j]  # Assuming batch size 1
            
            confidence = torch.sigmoid(pred[4]).item()
            
            if confidence > conf_threshold:
                # Convert grid coordinates to image coordinates
                x_center = (j + torch.sigmoid(pred[0])) / grid_size
                y_center = (i + torch.sigmoid(pred[1])) / grid_size
                width = torch.sigmoid(pred[2])
                height = torch.sigmoid(pred[3])
                
                # Convert to corner coordinates
                x1 = (x_center - width/2)
                y1 = (y_center - height/2)
                x2 = (x_center + width/2)
                y2 = (y_center + height/2)
                
                # Get class prediction
                class_probs = torch.softmax(pred[5:], dim=0)
                class_id = torch.argmax(class_probs).item()
                class_conf = class_probs[class_id].item()
                
                final_conf = confidence * class_conf
                
                if final_conf > conf_threshold:
                    boxes.append([x1.item(), y1.item(), x2.item(), y2.item()])
                    confidences.append(final_conf)
                    class_ids.append(class_id)
    
    if len(boxes) == 0:
        return [], [], []
    
    # Apply NMS
    boxes = np.array(boxes)
    confidences = np.array(confidences)
    class_ids = np.array(class_ids)
    
    indices = cv2.dnn.NMSBoxes(
        boxes.tolist(), 
        confidences.tolist(), 
        conf_threshold, 
        iou_threshold
    )
    
    if len(indices) > 0:
        indices = indices.flatten()
        return boxes[indices], confidences[indices], class_ids[indices]
    else:
        return [], [], []`}

            # Backward pass
            self.optimizer.zero_grad()
            loss.backward()
            torch.nn.utils.clip_grad_norm_(self.model.parameters(), max_norm=10.0)
            self.optimizer.step()
            
            total_loss += loss.item()
            num_batches += 1
            
            # Update progress bar
            progress_bar.set_postfix({
                'loss': f'{loss.item():.4f}',
                'coord': f'{loss_components["coord_loss"]:.4f}',
                'conf': f'{loss_components["conf_loss_obj"]:.4f}',
                'class': f'{loss_components["class_loss"]:.4f}'
            })
            
            # Log to TensorBoard
            global_step = epoch * len(train_loader) + batch_idx
            self.writer.add_scalar('Loss/Train_Step', loss.item(), global_step)
            
            for key, value in loss_components.items():
                self.writer.add_scalar(f'Loss_Components/{key}', value, global_step)
        
        avg_loss = total_loss / num_batches
        self.train_losses.append(avg_loss)
        
        return avg_loss
    
    def validate(self, val_loader, epoch):
        self.model.eval()
        total_loss = 0
        num_batches = 0
        
        with torch.no_grad():
            for images, targets in val_loader:
                images = images.to(self.device)
                targets = targets.to(self.device)
                
                predictions = self.model(images, mode='detection')
                loss, _ = self.criterion(predictions, targets, mode='detection')
                
                total_loss += loss.item()
                num_batches += 1
        
        avg_loss = total_loss / num_batches
        self.val_losses.append(avg_loss)
        
        # Log to TensorBoard
        self.writer.add_scalar('Loss/Validation', avg_loss, epoch)
        
        return avg_loss
    
    def train(self, train_loader, val_loader, num_epochs):
        print(f"Starting training for {num_epochs} epochs...")
        
        for epoch in range(num_epochs):
            # Training
            train_loss = self.train_epoch(train_loader, epoch)
            
            # Validation
            val_loss = self.validate(val_loader, epoch)
            
            # Learning rate scheduling
            self.scheduler.step(val_loss)
            
            print(f"Epoch {epoch+1}/{num_epochs}:")
            print(f"  Train Loss: {train_loss:.4f}")
            print(f"  Val Loss: {val_loss:.4f}")
            print(f"  LR: {self.optimizer.param_groups[0]['lr']:.2e}")
            
            # Save best model
            if val_loss < self.best_val_loss:
                self.best_val_loss = val_loss
                self.save_checkpoint('best_model.pth', epoch, val_loss)
                print(f"  New best model saved!")
            
            # Save regular checkpoint
            if (epoch + 1) % 10 == 0:
                self.save_checkpoint(f'checkpoint_epoch_{epoch+1}.pth', epoch, val_loss)
        
        self.writer.close()
        print("Training completed!")
    
    def save_checkpoint(self, filename, epoch, val_loss):
        checkpoint = {
            'epoch': epoch,
            'model_state_dict': self.model.state_dict(),
            'optimizer_state_dict': self.optimizer.state_dict(),
            'scheduler_state_dict': self.scheduler.state_dict(),
            'val_loss': val_loss,
            'train_losses': self.train_losses,
            'val_losses': self.val_losses
        }
        torch.save(checkpoint, filename)
